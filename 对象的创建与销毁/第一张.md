    @SuppressLint("UseSparseArrays")
    private Map<Integer, String> buttonText(OrderBottom order) {
        Map<Integer, String> map = new HashMap<>(2);
        switch (order.getO_state()) {
            case 0:
                map.put(1, "取消订单");
                map.put(2, "去支付");
                break;
            case 1:
                if (order.getOe_id() == 0) {
                    map.put(1, "");
                    map.put(2, "提醒发货");
                } else if (order.getOe_id() > 0) {
                    map.put(1, "查看物流");
                    map.put(2, "确认收货");
                }
                break;
            case 2:
                map.put(1, "取消订单");
                map.put(2, "核销使用");
                break;
            case 3:
                map.put(1, "查看详情");
                map.put(2, "评价");
                break;
            default:
                map.put(1, "");
                map.put(2, "查看详情");
                break;
        }
        return map;
    }
